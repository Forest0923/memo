    System Call #

    Use system-call crate #

    • Add a dependency to Cargo.toml:
    [dependencies]
system-call="0.1.3"
    • Sample code:
    #[macro_use]
extern crate syscall;

fn main() {
    let mut pid;
    let cpid;
    let mut val = 0;
    unsafe {
        pid = syscall!(GETPID);
    }
    println!("before:	val is {}, pid is {}", val, pid);
    unsafe {
        cpid = syscall!(FORK);
    }
    if cpid < 0 {
        println!("Fork error");
        return;
    } else if cpid == 0 {
        val += 1;
        unsafe {
            pid = syscall!(GETPID);
        }
        println!("child:	val is {}, pid is {}", val, pid);
        return;
    }
    unsafe {
        pid = syscall!(GETPID);
    }
    println!("parent:	val is {}, pid is {}", val, pid);
}
