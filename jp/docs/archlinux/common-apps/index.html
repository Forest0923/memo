    アプリケーションのインストール #

    Package Manager #

    • Paru
    git clone https://aur.archlinux.org/paru
cd paru
makepkg -si
    • Yay
    git clone https://aur.archlinux.org/yay.git
cd yay
makepkg -si

    Shell #

    • zsh
    sudo pacman -S zsh zsh-completions
chsh -s $(which zsh) # or chsh -s /bin/zsh

    Terminal Emulator #

    • Alacritty
    sudo pacman -S alacritty
    • Xfce4 terminal
    sudo pacman -S xfce4-terminal
    • Gnome terminal
    paru -S gnome-terminal-transparency

    Terminal multiplexer #

    • tmux
    sudo pacman -S tmux

    Editor #

    • VSCode
    paru -S visual-studio-code-bin
    • vim
    sudo pacman -S gvim
    • emacs
    sudo pacman -S emacs-nox

    Backup #

    • Timeshift
    paru -S timeshift

    Browser #

    • Brave browser
    paru -S brave-bin

    Virtual Machine Monitor #

    • KVM

    • VirtualBox

    sudo pacman -S virtualbox virtualbox-host-modules-arch
reboot

    Password Manager #

    • Bitwarden
    yay -S bitwarden-bin

    Anti Virus #

    • clamav
    sudo pacman -S clamav clamtk
sudo freshclam
sudo systemctl start clamav-freshclam.service
sudo systemctl enable clamav-freshclam.service
sudo systemctl start clamav-daemon.service
sudo systemctl enable clamav-daemon.service

    Firewall #

    • gufw
    sudo pacman -S gufw
sudo systemctl start ufw
sudo systemctl enable ufw
sudo ufw enable
    • firewalld
    sudo pacman -S firewalld

    Messaging Apps #

    • Discord
    paru -S discord
    • Slack
    paru -S slack-desktop
    • Quill
    sudo pacman -S quill-chat

    Office #

    • LibreOffice
    sudo pacman -S libreoffice

    rclone-browser #

    • rclone brawser
    sudo pacman -S rclone

    Gnome Shell #

    • CPU power manager
    sudo pacman -S gnome-shell-extension-cpupower-git
    • CPU temperature
    sudo pacman -S gnome-shell-extension-freon-git

    Preferences #

    • Gnome shell theme
    paru -S matcha-gtk-theme
    • Icon theme
    sudo pacman -S papirus-icon-theme breeze
    • Power line
    sudo pacman -S powerline powerline-fonts

    • Fonts

      • Japanese:
      sudo pacman -S otf-ipaexfont
yay -S ttf-ms-fonts
yay -S ttf-windows
      • Chinese & Hangul:
      sudo pacman -S adobe-source-han-sans-cn-fonts # Chinese
sudo pacman -S adobe-source-han-sans-tw-fonts # Chinese
sudo pacman -S adobe-source-han-sans-kr-fonts # hangul
